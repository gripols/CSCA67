\chapter*{Symbol Index}
\addcontentsline{toc}{chapter}{Symbol Index}

\begin{longtable}{llp{7cm}}
\toprule
\textbf{Symbol} & \textbf{Name} & \textbf{Meaning / Use} \\
\midrule
\multicolumn{3}{l}{\textbf{Logical Symbols}} \\
\midrule
$\neg P$ & Negation & “Not $P$.” The statement opposite in truth value to $P$. \\
$P \wedge Q$ & Conjunction & “$P$ and $Q$.” True only when both are true. \\
$P \vee Q$ & Disjunction & “$P$ or $Q$.” Inclusive ‘or’: true when at least one is true. \\
$P \to Q$ & Conditional / Implication & “If $P$, then $Q$.” False only when $P$ true, $Q$ false. \\
$P \leftrightarrow Q$ & Biconditional / Equivalence & “$P$ if and only if $Q$.” True when $P$ and $Q$ share truth values. \\
$\top$ & Tautology / Truth constant & Always true. \\
$\bot$ & Contradiction / Falsity & Always false. \\[3pt]

\multicolumn{3}{l}{\textbf{Quantifiers and Predicates}} \\
\midrule
$\forall x \in D$ & Universal quantifier & “For all $x$ in domain $D$.” \\
$\exists x \in D$ & Existential quantifier & “There exists an $x$ in $D$ such that….” \\
$P(x)$ & Predicate & A statement whose truth depends on variable $x$. \\
$D(x,y)$ & Binary predicate & A statement involving two variables, e.g. “$x$ divides $y$.” \\[3pt]

\multicolumn{3}{l}{\textbf{Set Theory}} \\
\midrule
$x \in A$ & Membership & $x$ is an element of set $A$. \\
$x \notin A$ & Non-membership & $x$ is not an element of set $A$. \\
$A \subseteq B$ & Subset & Every element of $A$ is also in $B$. \\
$A \subset B$ & Proper subset & $A \subseteq B$ but $A \neq B$. \\
$A \cup B$ & Union & Set of elements in $A$ or $B$ (or both). \\
$A \cap B$ & Intersection & Set of elements common to both $A$ and $B$. \\
$A \setminus B$ & Set difference & Elements in $A$ but not in $B$. \\
$A^\complement$ or $\overline{A}$ & Complement & Elements not in $A$, relative to the universe. \\
$A \triangle B$ & Symmetric difference & $(A \setminus B) \cup (B \setminus A)$ — elements in exactly one of $A$, $B$. \\
$\emptyset$ & Empty set & The set with no elements. \\
$\mathcal{P}(A)$ & Power set & The set of all subsets of $A$. \\[3pt]

\multicolumn{3}{l}{\textbf{Arithmetic and Number Theory}} \\
\midrule
$a \mid b$ & Divides & $a$ divides $b$; $\exists k \in \mathbb{Z}$ such that $b = ak$. \\
$a \nmid b$ & Does not divide & $a$ does not divide $b$. \\
$a \equiv b \pmod{n}$ & Congruence modulo $n$ & $n$ divides $(a - b)$. \\
$\mathbb{N}$ & Natural numbers & $\{1, 2, 3, \dots\}$ (sometimes includes $0$). \\
$\mathbb{Z}$ & Integers & $\{\dots, -2, -1, 0, 1, 2, \dots\}$. \\
$\mathbb{Q}$ & Rational numbers & Numbers of the form $\frac{p}{q}$ with $p, q \in \mathbb{Z}, q \neq 0$. \\
$\mathbb{R}$ & Real numbers & All real (continuous) numbers. \\
$\mathbb{C}$ & Complex numbers & Numbers of the form $a + bi$, where $i^2 = -1$. \\[3pt]

\multicolumn{3}{l}{\textbf{Proof and Logic Notation}} \\
\midrule
$\therefore$ & Therefore & Introduces a conclusion. \\
$\because$ & Because & Introduces a justification. \\
$\vdash$ & Derivability & Syntactic entailment: “provable that.” \\
$\models$ & Semantic entailment & “Logically entails.” True in all models. \\
$\iff$ & If and only if & Shortcut for $\leftrightarrow$. \\
$\Rightarrow$ & Implies & Often used in informal logical steps. \\
$\Leftrightarrow$ & Equivalent & Indicates logical equivalence. \\
$\Box$ & End of proof & Marks proof completion. \\[3pt]

\multicolumn{3}{l}{\textbf{Miscellaneous}} \\
\midrule
$\lvert A \rvert$ & Cardinality & Number of elements in $A$. \\
$\subseteq$, $\supseteq$ & Inclusion & “Is contained in” / “contains.” \\
$\land, \lor$ & Alternate connectives & Used in typed math instead of $\wedge, \vee$. \\
$\sim$ & Similarity / equivalence & Context-dependent relation symbol. \\
\bottomrule
\end{longtable}
